<script>
  const langBtn = document.getElementById('toggle-lang');
  let lang = 'fa';
  function setLang(newLang){
    lang = newLang;
    if(lang==='fa'){
      document.documentElement.lang='fa';
      document.getElementById('hero-title').innerText='بِنی واش — پاک‌کنندگیِ با اعتماد';
      document.getElementById('hero-desc').innerText='محصولات شویندهٔ تخصصی با تمرکز بر کیفیت، ایمنی الیاف و رایحه‌های لوکس.';
      langBtn.innerText='English';
      document.querySelector('header').style.flexDirection = 'row-reverse';
    } else {
      document.documentElement.lang='en';
      document.getElementById('hero-title').innerText='Benny Wash — Trusted Clean';
      document.getElementById('hero-desc').innerText='Premium cleaning products focused on performance, fabric care and luxurious scents.';
      langBtn.innerText='فارسی';
      document.querySelector('header').style.flexDirection = 'row';
    }
  }
  langBtn.addEventListener('click', ()=> setLang(lang==='fa'?'en':'fa'));
  setLang('fa');

  function openOrder(name, price){
    document.getElementById('order').style.display='block';
    document.getElementById('selected-product').innerText = name + ' — ' + price.toLocaleString() + ' تومان';
    window.scrollTo({top:document.getElementById('order').offsetTop-20, behavior:'smooth'});
  }
  function closeOrder(){ document.getElementById('order').style.display='none'; }

  function submitOrder(e){
    e.preventDefault();
    const name = document.getElementById('customer-name').value;
    const mobile = document.getElementById('customer-mobile').value;
    const email = document.getElementById('customer-email').value;
    const prod = document.getElementById('selected-product').innerText || '—';
    const qty = document.getElementById('qty').value;
    const subject = encodeURIComponent('سفارش جدید - Benny Wash');
    const body = encodeURIComponent('نام: '+name+'\\nموبایل: '+mobile+'\\nایمیل: '+email+'\\nمحصول: '+prod+'\\nتعداد: '+qty);
    // mailto as temporary order flow:
    window.location.href = 'mailto:Bennywash90@gmail.com?subject='+subject+'&body='+body;
    alert('سفارش شما آماده شد؛ ایمیل ارسال و برای پرداخت با شما تماس گرفته می‌شود.');
    closeOrder();
  }
</script>
